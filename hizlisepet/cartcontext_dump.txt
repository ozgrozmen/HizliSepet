锘縤mport { createContext, useContext, useState, useEffect } from 'react';
import { useAuth } from './AuthContext';
import { supabase } from '../lib/supabase';

const CartContext = createContext();

export function CartProvider({ children }) {
  const [cartItems, setCartItems] = useState([]);
  const [loading, setLoading] = useState(true);
  const { user } = useAuth();

  // Kullan脛卤c脛卤n脛卤n sepetini veritaban脛卤ndan 脙搂ek
  useEffect(() => {
    if (user) {
      fetchCartItems();
    } else {
      // Kullan脛卤c脛卤 giri脜鸥 yapmam脛卤脜鸥sa, yerel depolamadan sepeti y脙录kle
      const localCart = localStorage.getItem('cart');
      if (localCart) {
        setCartItems(JSON.parse(localCart));
      } else {
        setCartItems([]);
      }
      setLoading(false);
    }
  }, [user]);

  // Sepet de脛鸥i脜鸥ti脛鸥inde yerel depolamay脛卤 g脙录ncelle (giri脜鸥 yapmam脛卤脜鸥 kullan脛卤c脛卤lar i脙搂in)
  useEffect(() => {
    if (!user && cartItems.length > 0) {
      localStorage.setItem('cart', JSON.stringify(cartItems));
    }
  }, [cartItems, user]);

  const fetchCartItems = async () => {
    try {
      setLoading(true);
      
      const { data, error } = await supabase
        .from('cart')
        .select('*, products(*)')
        .eq('user_id', user.id);

      if (error) {
        console.error('Sepet y脙录klenirken hata:', error);
        setLoading(false);
        return;
      }

      setCartItems(data || []);
      setLoading(false);
    } catch (err) {
      console.error('Sepet y脙录klenirken beklenmeyen hata:', err);
      setLoading(false);
    }
  };

  const addToCart = async (product, quantity = 1, color = null, size = null) => {
    if (user) {
      // Oturum a脙搂m脛卤脜鸥 kullan脛卤c脛卤lar i脙搂in veritaban脛卤na kaydet
      try {
        // 脙鈥搉ce ayn脛卤 脙录r脙录n脙录n sepette olup olmad脛卤脛鸥脛卤n脛卤 kontrol et
        const existingItemIndex = cartItems.findIndex(
          item => item.product_id === product.id && 
                 item.color === color && 
                 item.size === size
        );

        if (existingItemIndex >= 0) {
          // 脙艙r脙录n zaten sepette, miktar脛卤 g脙录ncelle
          const updatedItems = [...cartItems];
          const newQuantity = updatedItems[existingItemIndex].quantity + quantity;

          const { error } = await supabase
            .from('cart')
            .update({ quantity: newQuantity })
            .eq('id', updatedItems[existingItemIndex].id);

          if (error) {
            console.error('Sepet g脙录ncellenirken hata:', error);
            return false;
          }

          updatedItems[existingItemIndex].quantity = newQuantity;
          setCartItems(updatedItems);
        } else {
          // Yeni 脙录r脙录n ekle
          const { data, error } = await supabase
            .from('cart')
            .insert([
              { 
                user_id: user.id, 
                product_id: product.id, 
                quantity, 
                color, 
                size,
                price: product.price
              }
            ])
            .select('*, products(*)');

          if (error) {
            console.error('Sepete eklenirken hata:', error);
            return false;
          }

          setCartItems([...cartItems, ...data]);
        }

        return true;
      } catch (err) {
        console.error('Sepete ekleme hatas脛卤:', err);
        return false;
      }
    } else {
      // Oturum a脙搂mam脛卤脜鸥 kullan脛卤c脛卤lar i脙搂in yerel depolamada sakla
      const existingItemIndex = cartItems.findIndex(
        item => item.product_id === product.id && 
               item.color === color && 
               item.size === size
      );

      if (existingItemIndex >= 0) {
        const updatedItems = [...cartItems];
        updatedItems[existingItemIndex].quantity += quantity;
        setCartItems(updatedItems);
      } else {
        setCartItems([
          ...cartItems,
          {
            id: Date.now().toString(),
            product_id: product.id,
            product: product,
            quantity,
            color,
            size,
            price: product.price
          }
        ]);
      }
      
      return true;
    }
  };

  const removeFromCart = async (itemId) => {
    if (user) {
      // Oturum a脙搂m脛卤脜鸥 kullan脛卤c脛卤lar i脙搂in veritaban脛卤ndan sil
      try {
        const { error } = await supabase
          .from('cart')
          .delete()
          .eq('id', itemId);

        if (error) {
          console.error('Sepetten silinirken hata:', error);
          return false;
        }

        setCartItems(cartItems.filter(item => item.id !== itemId));
        return true;
      } catch (err) {
        console.error('Sepetten silme hatas脛卤:', err);
        return false;
      }
    } else {
      // Oturum a脙搂mam脛卤脜鸥 kullan脛卤c脛卤lar i脙搂in yerel depolamadan sil
      setCartItems(cartItems.filter(item => item.id !== itemId));
      return true;
    }
  };

  const updateQuantity = async (itemId, newQuantity) => {
    if (newQuantity <= 0) {
      return removeFromCart(itemId);
    }

    if (user) {
      // Oturum a脙搂m脛卤脜鸥 kullan脛卤c脛卤lar i脙搂in veritaban脛卤nda g脙录ncelle
      try {
        const { error } = await supabase
          .from('cart')
          .update({ quantity: newQuantity })
          .eq('id', itemId);

        if (error) {
          console.error('Miktar g脙录ncellenirken hata:', error);
          return false;
        }

        setCartItems(
          cartItems.map(item => 
            item.id === itemId ? { ...item, quantity: newQuantity } : item
          )
        );
        return true;
      } catch (err) {
        console.error('Miktar g脙录ncelleme hatas脛卤:', err);
        return false;
      }
    } else {
      // Oturum a脙搂mam脛卤脜鸥 kullan脛卤c脛卤lar i脙搂in yerel depolamada g脙录ncelle
      setCartItems(
        cartItems.map(item => 
          item.id === itemId ? { ...item, quantity: newQuantity } : item
        )
      );
      return true;
    }
  };

  const clearCart = async () => {
    if (user) {
      // Oturum a脙搂m脛卤脜鸥 kullan脛卤c脛卤lar i脙搂in veritaban脛卤ndan sepeti temizle
      try {
        const { error } = await supabase
          .from('cart')
          .delete()
          .eq('user_id', user.id);

        if (error) {
          console.error('Sepet temizlenirken hata:', error);
          return false;
        }

        setCartItems([]);
        return true;
      } catch (err) {
        console.error('Sepet temizleme hatas脛卤:', err);
        return false;
      }
    } else {
      // Oturum a脙搂mam脛卤脜鸥 kullan脛卤c脛卤lar i脙搂in yerel depolamay脛卤 temizle
      setCartItems([]);
      localStorage.removeItem('cart');
      return true;
    }
  };

  const getCartTotal = () => {
    return cartItems.reduce((total, item) => {
      // 脙艙r脙录n fiyat脛卤n脛卤 al脛卤rken, 脙录r脙录n bilgisi farkl脛卤 formatta olabilir
      const price = item.price || (item.product ? item.product.price : 0);
      return total + (price * item.quantity);
    }, 0);
  };

  const getCartItemCount = () => {
    return cartItems.reduce((total, item) => total + item.quantity, 0);
  };

  return (
    <CartContext.Provider
      value={{
        cartItems,
        loading,
        addToCart,
        removeFromCart,
        updateQuantity,
        clearCart,
        getCartTotal,
        getCartItemCount
      }}
    >
      {children}
    </CartContext.Provider>
  );
}

export function useCart() {
  return useContext(CartContext);
}
